```ini
.env
ALBINA_LOLA_KRONOS_API_TOKEN=
ALBINA_LWDKIP_PASSWORD=

ALBINA_WIKISNOW_API=
MYSQL_HOST=
MYSQL_PORT=
MYSQL_USER=
MYSQL_PASSWORD=
MYSQL_DATABASE=

```

```sh
> corepack enable
> yarn install
> yarn dev
```
